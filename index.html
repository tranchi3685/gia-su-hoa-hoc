var lastRow = sheet.getLastRow(); function tinhDiem(soLuot) { if (soLuot >= 15) return 2; if (soLuot >= 10) return 1.5; if (soLuot >= 5) return 1; if (soLuot >= 3) return 0.5; return 0; } if (lastRow >= 2) { var data = sheet.getRange(2, 1, lastRow - 1, 4).getValues(); for (var i = 0; i < data.length; i++) { if (String(data[i][0]) === userId && String(data[i][1]) === String(tuan)) { var soLuot = Number(data[i][2]) + 1; sheet.getRange(i + 2, 3).setValue(soLuot); sheet.getRange(i + 2, 4).setValue(tinhDiem(soLuot)); return; } } } // Chưa có → thêm dòng mới sheet.appendRow([userId, tuan, 1, tinhDiem(1), '']); } catch(e) { Logger.log("Lỗi updateTienDo: " + e.toString()); } } function logNgoaiLuong(userId, text) { try { var ss = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID);
